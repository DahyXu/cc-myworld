# 地图标注（Waypoint）Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 大地图点击设置单个标注点，游戏内底部居中显示旋转方向箭头（指向标注点 + 距离），可通过大地图 ✕ 按钮取消；小地图上也显示标注点；localStorage 持久化。

**Architecture:** 纯客户端，零服务器改动。全部逻辑在 `js/minimap.js` IIFE 内新增私有函数和状态；`index.html` 新增 `#waypointHud` canvas；`update()` 每帧驱动 HUD 箭头绘制。

**Tech Stack:** HTML5 Canvas 2D API，localStorage，已有 IIFE 模块模式（`MyWorld.Minimap`）

**Spec:** `docs/superpowers/specs/2026-06-15-waypoint-design.md`

```js
  function _onMapClick(e) {
    if (!mapOpen) return;
    const rect = mapCanvas.getBoundingClientRect();
    const mx = e.clientX - rect.left;
    const my = e.clientY - rect.top;
    if (waypointXBtn && Math.hypot(mx - waypointXBtn.x, my - waypointXBtn.y) <= waypointXBtn.r) {
      _clearWaypoint();
      return;
    }
    const W = mapCanvas.width, H = mapCanvas.height;
    _setWaypoint(mx / W * WX_RANGE + WX_MIN, my / H * WZ_RANGE + WZ_MIN);
  }

  function _setWaypoint(wx, wz) {
    waypoint = { wx, wz };
    localStorage.setItem('waypoint', JSON.stringify(waypoint));
  }

  function _clearWaypoint() {
    waypoint = null;
    waypointXBtn = null;
    localStorage.removeItem('waypoint');
  }

  function _drawStar(ctx, cx, cy, outerR, innerR, color) {
    ctx.beginPath();
    for (let i = 0; i < 10; i++) {
      const r = i % 2 === 0 ? outerR : innerR;
      const a = (Math.PI / 5) * i - Math.PI / 2;
      i === 0 ? ctx.moveTo(cx + r * Math.cos(a), cy + r * Math.sin(a))
              : ctx.lineTo(cx + r * Math.cos(a), cy + r * Math.sin(a));
    }
    ctx.closePath();
    ctx.fillStyle = color;
    ctx.fill();
  }

  function _roundRect(ctx, x, y, w, h, r) {
    ctx.beginPath();
    ctx.moveTo(x + r, y);
    ctx.lineTo(x + w - r, y);
    ctx.arcTo(x + w, y, x + w, y + r, r);
    ctx.lineTo(x + w, y + h - r);
    ctx.arcTo(x + w, y + h, x + w - r, y + h, r);
    ctx.lineTo(x + r, y + h);
    ctx.arcTo(x, y + h, x, y + h - r, r);
    ctx.lineTo(x, y + r);
    ctx.arcTo(x, y, x + r, y, r);
    ctx.closePath();
  }
```

```js
  function _drawWaypointHud(player) {
    if (!waypoint || !player) {
      waypointHudEl.style.display = 'none';
      return;
    }
    waypointHudEl.style.display = 'block';
    const ctx = wpCtx;
    ctx.clearRect(0, 0, 120, 44);

    // 背景
    ctx.fillStyle = 'rgba(0,0,0,0.62)';
    _roundRect(ctx, 0, 0, 120, 44, 8);
    ctx.fill();

    // 相对方位角（玩家坐标系）
    // 前向 (fx,fz) = (-sin(yaw), -cos(yaw))
    // 右向 = 前向顺时针90°: (cos(yaw), -sin(yaw))
    const dx = waypoint.wx - player.x;
    const dz = waypoint.wz - player.z;
    const dotF = -(dx * Math.sin(player.yaw) + dz * Math.cos(player.yaw));
    const dotR =   dx * Math.cos(player.yaw) - dz * Math.sin(player.yaw);
    const rel = Math.atan2(dotR, dotF);  // 0=正前 π/2=右 -π/2=左

    // 旋转箭头（左侧 44×44 区域，圆心 (22,22)）
    ctx.save();
    ctx.translate(22, 22);
    ctx.rotate(rel);
    ctx.beginPath();
    ctx.moveTo(0, -13);
    ctx.lineTo(7, 7);
    ctx.lineTo(0, 3);
    ctx.lineTo(-7, 7);
    ctx.closePath();
    ctx.fillStyle = '#ffe033';
    ctx.fill();
    ctx.restore();

    // 距离文字
    const dist = Math.hypot(dx, dz);
    ctx.fillStyle = '#ffe033';
    ctx.font = 'bold 13px sans-serif';
    ctx.textAlign = 'left';
    ctx.textBaseline = 'middle';
    ctx.fillText(Math.round(dist) + ' 格', 48, 22);
  }
```
